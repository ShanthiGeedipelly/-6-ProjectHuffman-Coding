#include <bits/stdc++.h>
using namespace std;

// Huffman Tree Node
struct Node {
    char data;
    int freq;
    Node *left, *right;

    Node(char d, int f) {
        data = d;
        freq = f;
        left = right = NULL;
    }
};

// Comparator for priority queue
struct Compare {
    bool operator()(Node* a, Node* b) {
        return a->freq > b->freq;
    }
};

// Preorder traversal to print Huffman codes
void printCodes(Node* root, string code) {
    if (!root)
        return;

    // Leaf node
    if (!root->left && !root->right) {
        cout << code << " ";
    }

    printCodes(root->left, code + "0");
    printCodes(root->right, code + "1");
}

// Main Huffman function
void huffmanCodes(string S, vector<int> f) {
    priority_queue<Node*, vector<Node*>, Compare> pq;

    // Create leaf nodes
    for (int i = 0; i < S.size(); i++) {
        pq.push(new Node(S[i], f[i]));
    }

    // Build Huffman Tree
    while (pq.size() > 1) {
        Node* left = pq.top(); pq.pop();
        Node* right = pq.top(); pq.pop();

        Node* parent = new Node('$', left->freq + right->freq);
        parent->left = left;
        parent->right = right;

        pq.push(parent);
    }

    // Print Huffman Codes in preorder
    printCodes(pq.top(), "");
}

// Driver code
int main() {
    string S = "abcdef";
    vector<int> f = {5, 9, 12, 13, 16, 45};

    huffmanCodes(S, f);
    return 0;
}
